ALGORITHM binary_search(arr, key, begin, end):
    '''
    INPUT :
            arr - a sequence of sorted integers;
            key - integer to search for;
            begin - start index of the sequence;
            end - ending index of the sequence
    OUTPUT:
            index of the element or key IF found
            -1 IF not found
    '''
1      IF end >= begin:
2          SET middle TO (begin + end) // 2
3          IF arr[middle] EQUALS key:
4              RETURN middle
5          ELSEIF arr[middle] > key:
6              RETURN binary_search(arr, key, begin, middle)   # lower half
7          ELSE:
8              RETURN binary_search(arr, key, middle + 1, end)  # upper half
9  
10      RETURN -1  # key not found
